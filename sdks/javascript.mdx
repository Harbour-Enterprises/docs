## Quick Start

<CodeGroup>
```javascript ES6 Modules
import { SuperDoc } from '@superdoc/api';

const client = new SuperDoc('sd_live_pk_your_api_key');

// Convert a local DOCX file to PDF
const pdfBuffer = await client.convert({
file: './document.docx',
format: 'pdf'
});

// Save the PDF
import fs from 'fs';
fs.writeFileSync('output.pdf', pdfBuffer);

````

```javascript CommonJS
const { SuperDoc } = require('@superdoc/api');

const client = new SuperDoc('sd_live_pk_your_api_key');

async function convertDocument() {
  try {
    const pdfBuffer = await client.convert({
      file: './document.docx',
      format: 'pdf'
    });

    require('fs').writeFileSync('output.pdf', pdfBuffer);
    console.log('Conversion successful!');
  } catch (error) {
    console.error('Conversion failed:', error.message);
  }
}

convertDocument();
````


## Configuration

### Basic Configuration

```javascript
const client = new SuperDoc("sd_live_pk_your_api_key", {
  baseURL: "https://api.superdoc.dev", // Custom API endpoint
  timeout: 60000, // Request timeout in ms
  retries: 3, // Number of retry attempts
  retryDelay: 1000, // Initial retry delay in ms
});
```

### Advanced Configuration

```javascript
const client = new SuperDoc("sd_live_pk_your_api_key", {
  // API Configuration
  baseURL: "https://api.superdoc.dev",
  timeout: 60000,

  // Retry Configuration
  retries: 3,
  retryDelay: 1000,
  retryBackoff: "exponential", // 'exponential' or 'linear'

    protocol: "http",
    host: "proxy.example.com",
    port: 8080,
    auth: {
      username: "proxy_user",
      password: "proxy_pass",
    },
  },

  // Logging
  debug: true, // Enable debug logging
  logger: console, // Custom logger instance
});
```

## API Methods

### convert()

Convert a DOCX file to PDF format.

```javascript
const result = await client.convert(options);
```

#### Parameters

<ResponseField name="options" type="object" required>
  Conversion options object
  <Expandable title="options properties">
    <ResponseField
      name="file"
      type="string | File | Buffer | ReadableStream"
      required
    >
      File to convert. Can be a file path (Node.js), File object (browser),
      Buffer, or ReadableStream
    </ResponseField>

    <ResponseField name="format" type="string">
      Output format. Currently only 'pdf' is supported. Defaults to 'pdf'
    </ResponseField>

    <ResponseField name="filename" type="string">
      Original filename for better error reporting and metadata
    </ResponseField>

    <ResponseField name="onProgress" type="function">
      Progress callback function: `(progress: ProgressEvent) => void`
    </ResponseField>
  </Expandable>
</ResponseField>

#### Returns

<ResponseField name="result" type="Buffer | Blob">
  - **Node.js**: Returns a Buffer containing the PDF data - **Browser**: Returns
  a Blob containing the PDF data
</ResponseField>

## Framework Integration

### Express.js

```javascript
const express = require("express");
const multer = require("multer");
const { SuperDoc } = require("@superdoc/api");

const app = express();
const upload = multer();
const client = new SuperDoc(process.env.SUPERDOC_API_KEY);

app.post("/convert", upload.single("file"), async (req, res) => {
  try {
    if (!req.file) {
      return res.status(400).json({ error: "No file provided" });
    }

    const pdf = await client.convert({
      file: req.file.buffer,
      format: "pdf",
      filename: req.file.originalname,
    });

    res.setHeader("Content-Type", "application/pdf");
    res.setHeader(
      "Content-Disposition",
      'attachment; filename="converted.pdf"'
    );
    res.send(pdf);
  } catch (error) {
    console.error("Conversion error:", error);
    res.status(500).json({
      error: error.message,
      code: error.code,
      requestId: error.requestId,
    });
  }
});
```

### Next.js API Route

```javascript
// pages/api/convert.js
import { SuperDoc } from "@superdoc/api";
import formidable from "formidable";
import fs from "fs";

const client = new SuperDoc(process.env.SUPERDOC_API_KEY);

export default async function handler(req, res) {
  if (req.method !== "POST") {
    return res.status(405).json({ error: "Method not allowed" });
  }

  const form = formidable();
  const [fields, files] = await form.parse(req);
  const file = files.file[0];

  try {
    const pdf = await client.convert({
      file: file.filepath,
      format: "pdf",
      filename: file.originalFilename,
    });

    res.setHeader("Content-Type", "application/pdf");
    res.send(pdf);
  } catch (error) {
    res.status(500).json({
      error: error.message,
      code: error.code,
    });
  } finally {
    // Clean up temporary file
    if (file.filepath) {
      fs.unlinkSync(file.filepath);
    }
  }
}

export const config = {
  api: {
    bodyParser: false,
  },
};
```

### React Hook

```javascript
import { useState, useCallback } from "react";
import { SuperDoc } from "@superdoc/api";

const useDocumentConverter = (apiKey) => {
  const [isConverting, setIsConverting] = useState(false);
  const [progress, setProgress] = useState(0);
  const [error, setError] = useState(null);

  const convert = useCallback(
    async (file) => {
      setIsConverting(true);
      setProgress(0);
      setError(null);

      try {
        // In production, never expose API keys in the browser
        // This should be done through your backend API
        const client = new SuperDoc(apiKey);

        const pdf = await client.convert({
          file,
          format: "pdf",
          onProgress: (progressEvent) => {
            setProgress(progressEvent.percentage);
          },
        });

        // Create download link
        const url = URL.createObjectURL(pdf);
        const a = document.createElement("a");
        a.href = url;
        a.download = "converted.pdf";
        a.click();
        URL.revokeObjectURL(url);

        return pdf;
      } catch (err) {
        setError(err);
        throw err;
      } finally {
        setIsConverting(false);
        setProgress(0);
      }
    },
    [apiKey]
  );

  return { convert, isConverting, progress, error };
};

// Usage
function DocumentConverter() {
  const { convert, isConverting, progress, error } = useDocumentConverter();

  const handleFileSelect = async (event) => {
    const file = event.target.files[0];
    if (file) {
      try {
        await convert(file);
      } catch (error) {
        console.error("Conversion failed:", error);
      }
    }
  };

  return (
    <div>
      <input type="file" accept=".docx" onChange={handleFileSelect} />
      {isConverting && <div>Converting... {Math.round(progress)}%</div>}
      {error && <div>Error: {error.message}</div>}
    </div>
  );
}
```

## Advanced Usage

### Custom Retry Logic

```javascript
const client = new SuperDoc("sd_live_pk_your_api_key", {
  retries: 0, // Disable built-in retries
});

async function convertWithCustomRetry(file, maxRetries = 3) {
  for (let attempt = 1; attempt <= maxRetries; attempt++) {
    try {
      return await client.convert({ file, format: "pdf" });
    } catch (error) {
      if (attempt === maxRetries) throw error;

      if (error instanceof RateLimitError) {
        // Wait for rate limit reset
        await new Promise((resolve) =>
          setTimeout(resolve, error.retryAfter * 1000)
        );
      } else if (error.status >= 500) {
        // Exponential backoff for server errors
        const delay = Math.pow(2, attempt - 1) * 1000;
        await new Promise((resolve) => setTimeout(resolve, delay));
      } else {
        // Don't retry client errors
        throw error;
      }
    }
  }
}
```

### Batch Processing

```javascript
async function convertBatch(files, concurrency = 3) {
  const results = [];
  const errors = [];

  // Process files in batches to respect rate limits
  for (let i = 0; i < files.length; i += concurrency) {
    const batch = files.slice(i, i + concurrency);

    const batchPromises = batch.map(async (file, index) => {
      try {
        const pdf = await client.convert({ file, format: "pdf" });
        return { success: true, file, pdf, index: i + index };
      } catch (error) {
        errors.push({ file, error, index: i + index });
        return { success: false, file, error, index: i + index };
      }
    });

    const batchResults = await Promise.allSettled(batchPromises);
    results.push(...batchResults.map((r) => r.value || r.reason));

    // Add delay between batches if needed
    if (i + concurrency < files.length) {
      await new Promise((resolve) => setTimeout(resolve, 1000));
    }
  }

  return { results, errors };
}
```

## Troubleshooting

<AccordionGroup>
  <Accordion title="Module not found errors" icon="exclamation">
    **Problem**: `Cannot find module '@superdoc/api'`
    
    **Solution**: 
    - Ensure the package is installed: `npm install @superdoc/api`
    - Check your Node.js version (requires Node.js 14+)
    - Clear npm cache: `npm cache clean --force`
  </Accordion>

{" "}
<Accordion title="Authentication errors" icon="lock">
  **Problem**: `401 Unauthorized` or `Invalid API key` **Solution**: - Verify
  your API key is correct - Check environment variables are loaded - Ensure API
  key has proper permissions - Test with a fresh API key
</Accordion>

{" "}
<Accordion title="File upload issues" icon="upload">
  **Problem**: Files fail to upload or convert **Solution**: - Check file size
  limits (25MB for authenticated, 5MB for public) - Ensure file is a valid DOCX
  format - Verify file is not corrupted - Check network connectivity
</Accordion>

  <Accordion title="Browser compatibility" icon="browser">
    **Problem**: SDK doesn't work in browser
    
    **Solution**:
    - Use a bundler (Webpack, Rollup, etc.)
    - For production, proxy requests through your backend
    - Ensure polyfills for older browsers if needed
    - Check CORS policy compliance
  </Accordion>
</AccordionGroup>
